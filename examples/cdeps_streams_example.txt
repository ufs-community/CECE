# CDEPS Streams Configuration Example for ACES
# This file demonstrates the ESMF Config format for CDEPS-inline data ingestion
#
# Format:
#   stream::<stream_name>
#     file_paths = <comma-separated list of NetCDF files>
#     variables = <var_in_file:var_in_model, ...>
#     taxmode = <cycle|extend|limit>
#     tintalgo = <none|linear|nearest|lower|upper>
#     mapalgo = <bilinear|patch|...>
#     yearFirst = <first year in data>
#     yearLast = <last year in data>
#     yearAlign = <year to align with model time>
#     offset = <time offset in seconds>
#   ::

# Example 1: Anthropogenic emissions from CEDS inventory
stream::ceds_anthro
  file_paths = /data/emissions/CEDS_CO_anthro_2020.nc, /data/emissions/CEDS_NOx_anthro_2020.nc
  variables = CO_emis:CEDS_CO, NOx_emis:CEDS_NOx
  taxmode = cycle
  tintalgo = linear
  mapalgo = bilinear
  yearFirst = 2020
  yearLast = 2020
  yearAlign = 2020
  offset = 0
::

# Example 2: Biogenic emissions from MEGAN
stream::megan_biogenic
